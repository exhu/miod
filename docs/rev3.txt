type MyFloat = float32

proc MyAbs(a: MyFloat): MyFloat do
    return fabs(a)
endp

construction. http://www.freepascal.org/docs-html/ref/refsu65.html

-3.5..4.5 -> float64
-3.5f..4.5f -> float32, 64
